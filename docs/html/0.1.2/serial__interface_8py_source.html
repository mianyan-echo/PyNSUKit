20def head_check(send_cmd, recv_cmd):
21 """!
22 @brief 包头检查
23 @details 返回数据包头检查
24 @param send_cmd 发送的指令
25 @param recv_cmd 返回的指令
26 @return 返回指令的总长度
27 """
28 send_head = struct.unpack('=IIII', send_cmd[0:16])
29 recv_head = struct.unpack('=IIII', recv_cmd)
30 if recv_head[0] != 0xCFCFCFCF:
31 raise RuntimeError("返回包头错误")
32 if recv_head[1] != send_head[1]:
33 raise RuntimeError("返回ID错误")
34 if recv_head[2] != send_head[2]:
35 raise RuntimeError("返回序号错误")
36 return recv_head[3]

39class SerialCmdUItf(BaseCmdUItf):
40 """!
41 @brief 串口指令接口
42 @details 包括连接/断开、发送、接收等功能
43 @image html professional_serial_cmd.png
44 """
45 _target = 'COM0'
46 _target_baud_rate = 9600
47 _timeout = 15

49 def __init__(self):
50 self._device_serial = None
51 self.busy_lock = Lock()

53 def accept(self, target=None, target_baud_rate: int = None, **kwargs):
54 """!
55 @brief 初始化串口指令接口
56 @details 初始化串口指令接口，获取串口id，波特率等参数
57 @param target 串口id
58 @param target_baud_rate 波特率
59 @param kwargs 其他参数
60 @return
61 """
62 _target = self._target if target is None else target
63 _target_baud_rate = self._target_baud_rate if target_baud_rate is None else target_baud_rate
64 with self.busy_lock:
65 if self._device_serial is not None:
66 self._device_serial.close()
67 self._device_serial = serial.Serial(port=_target,
68 baudrate=int(_target_baud_rate),
69 timeout=self._timeout)

71 def recv_bytes(self, size) -> bytes:
72 """!
73 @brief 接收数据
74 @details 使用串口接收指定大小的数据
75 @param size 接收数据的长度
76 @return 接收到的数据
77 """
78 with self.busy_lock:
79 recv_data = b''
80 recv_size = 0
81 while True:
82 if recv_size != size:
83 data = self._device_serial.read(size)
84 recv_data += data
85 recv_size += len(data)
86 if recv_size >= size:
87 break
88 return recv_data

90 def send_bytes(self, data: bytes):
91 """!
92 @brief 发送数据
93 @details 使用串口发送数据
94 @param data 要发送的数据
95 @return 发送完成的数据长度
96 """
97 with self.busy_lock:
98 total_len = len(data)
99 total_sendlen = 0
100 while True:
101 send_len = self._device_serial.write(data)
102 total_sendlen += send_len
103 if total_len == total_sendlen:
104 return total_len
105 if send_len == 0:
106 raise RuntimeError("Connection interruption")

108 def write(self, addr: int, value: bytes) -> bytes:
109 """!
110 @brief 发送数据
111 @details 使用串口以地址值的方式发送一条约定好的特殊指令
112 @param addr 要修改的地址
113 @param value 地址中要赋的值
114 @return 返回数据中的结果
115 """
116 cmd = self._fmt_reg_write(addr, value)
117 if len(cmd) != self.send_bytessend_bytes(cmd):
118 raise RuntimeError(f"Fail in send")
119 recv = self.recv_bytesrecv_bytes(16)
120 result_len = head_check(cmd, recv)
121 result = self.recv_bytesrecv_bytes(result_len - 16)
122 return result
