29def dispenser(func):
30 """!
31 装饰器，让被装饰的函数根据stream_mode将调用分到指定的实现上
32 @param func:
33 @return:
34 """
35
36 @wraps(func)
37 def wrapper(*args, **kwargs):
38 self: "VirtualChnlMw" = args[0]
39 if self.stream_mode == self.StreamMode.VIRTUAL:
40 _func = func
41 else:
42 _func = getattr(self.kit.itf_chnl, func.__name__)
43 return _func(*args, **kwargs)
44
45 return wrapper
