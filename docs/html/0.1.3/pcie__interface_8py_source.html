126 def send_bytes(self, data: bytes):
127 """!
128 @brief icd指令使用pcie发送
129 @details 只有在使用icd_parser发送指令时会用
130 @param data 要发送的数据
131 @return 已发送的数据长度
132 """
133 try:
134 self.sent_ptr = 0
135 total_length, sent_length = len(data), 0
136 st = time.time()
137 while total_length != sent_length:
138 assert not self._break_status(), "Interrupt command reception"
139 sent_length += self._send(data[sent_length: self._block_size + sent_length])
140 assert time.time() - st < self.timeout, f"send timeout, sent {sent_length}"
141 self.sent_downsent_downsent_down = True
142 self.timeout -= (time.time() - st)
143 return sent_length
144 except AssertionError as e:
145 assert 0, f"[toaxi] {e}"

147 def recv_bytes(self, bufsize: int):
148 """!
149 @brief icd指令使用pcie接收
150 @details 只有在使用icd_parser接收指令时会用
151 @param bufsize 要接收数据的长度
152 @return 接收到的数据
153 """
154 try:
155 self.recv_ptr = 0
156 if bufsize != 0:
157 if self.wait_irq:
158 self.per_recv()
159 else:
160 self.per_recv_polled()
161 block_size, bytes_data, bytes_data_length = self._block_size, b"", 0
162 st = time.time()
163 while bytes_data_length != bufsize:
164 assert not self._break_status(), "Interrupt command reception"
165 if not (bufsize - bytes_data_length) // self._block_size:
166 block_size = (bufsize - bytes_data_length) % self._block_size
167 cur_recv_data = self._recv(block_size)
168 bytes_data += cur_recv_data
169 bytes_data_length += len(cur_recv_data)
170 assert time.time() - st < self.timeout, f"recv timeout, rcvd {bytes_data_length}"
171 except (AssertionError, TimeoutError) as e:
172 assert 0, f"[toaxi] {e}"
173 return bytes_data

175 def _send(self, data):
176 """!
177 @brief 指令发送
178 @details 数据不满足4Bytes整倍数的，被自动补齐为4Bytes整倍数，使用wr_lite写入数据
179 @param data 要发送的数据
180 @return 已经发送的数据长度
181 """
182 data += b'\x00' * (len(data) % 4)
183 data = np.frombuffer(data, dtype=np.uint32)
184 for value in data:
185 self.xdma.alite_write(self.sent_base + self.sent_ptr, value, self.board)
186 self.sent_ptr += 4
187 return int(data.nbytes)
188
189 def _recv(self, size):
190 """!
191 @brief 指令接收
192 @details 使用rd_lite读取数据
193 @param size 要接收的数据大小
194 @return 接收的数据
195 """
196 recv_size = size + size % 4
197 recv_size //= 4
198 res = np.zeros((recv_size,), dtype=np.uint32)
199 for idx in range(recv_size):
200 res[idx] = self.xdma.alite_read(self.recv_base + self.recv_ptr, self.board)[1]
201 self.recv_ptr += 4
202 return res.tobytes()[:size]

204 @property
205 def sent_down(self):
206 return self.xdma.alite_read(self.sent_base + self.ADDR_SENT_DOWN * 4, self.board)[1]

210 """!
211 @brief 标识

215 if value:

225 """

230 def per_recv(self, callback=None):
231 """!
232 @brief 接收数据前
233 @details 在接收数据前运行，等待数据准备完成
234 @param callback 回调函数
235 @return
236 """
237 res = self.xdma.wait_irq(self.irq_num, self.board, self.timeout * 1000)
238 if not res:
239 raise TimeoutError(f'toaxi timeout')
240 if callable(callback):
241 callback()
242 self.reset_irq()
243 self.recv_event.set()
244
245 def per_recv_polled(self):
246 timeout = self.timeout
247 while True:
248 if self.xdma.alite_read(self.irq_base, self.board)[1] != 0x8000:
249 time.sleep(1)
250 if timeout > 0:
251 timeout -= 1
252 else:
253 raise TimeoutError(f'toaxi timeout')
254 else:
255 self.reset_irq()
256 self.recv_event.set()
257 break
