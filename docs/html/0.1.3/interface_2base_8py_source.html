54class BaseCmdUItf(UInterface):
55 @staticmethod
56 def _fmt_reg_write(reg: int = 0, value: bytes = b'') -> bytes:
57 """!
58 @brief 格式化TCP/serial模拟写寄存器功能的icd
59 @param reg: 寄存器地址
60 @param value: 寄存器值
61 @return 格式化好的icd指令
62 """
63 if not isinstance(value, bytes):
64 raise RuntimeError("The value not be pack")
65 pack = (0x5F5F5F5F, 0x31001000, 0x00000000, 24, reg)
66 send_cmd = struct.pack('=IIIII', *pack)
67 send_cmd += value
68 return send_cmd
