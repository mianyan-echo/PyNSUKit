27def head_check(send_cmd, recv_cmd):
28 """!
29 @brief 包头检查
30 @details 返回数据包头检查
31 @param send_cmd 发送的指令
32 @param recv_cmd 返回的指令
33 @return 返回指令的总长度
34 """
35 send_head = struct.unpack('=IIII', send_cmd[0:16])
36 recv_head = struct.unpack('=IIII', recv_cmd)
37 if recv_head[0] != 0xCFCFCFCF:
38 raise RuntimeError("返回包头错误")
39 if recv_head[1] != send_head[1]:
40 raise RuntimeError("返回ID错误")
41 if recv_head[2] != send_head[2]:
42 raise RuntimeError("返回序号错误")
43 return recv_head[3]

75 def recv_bytes(self, size: int) -> bytes:
76 """!
77 @brief 接收数据
78 @details 使用网络接收指定大小的数据
79 @param size 接收数据的长度
80 @return 接收到的数据
81 """
82 with self.busy_lock:
83 recv_data = b''
84 recv_size = 0
85 while True:
86 if recv_size != size:
87 data = self._tcp_server.recv(size-recv_size)
88 recv_data += data
89 recv_size += len(data)
90 if recv_size >= size:
91 break
92 return recv_data

94 def send_bytes(self, data: bytes) -> int:
95 """!
96 @brief 发送数据
97 @details 使用网络发送数据
98 @param data 要发送的数据
99 @return 发送完成的数据长度
100 """
101 with self.busy_lock:
102 total_len = len(data)
103 total_sendlen = 0
104 while True:
105 send_len = self._tcp_server.send(data)
106 total_sendlen += send_len
107 if total_len == total_sendlen:
108 return total_len
109 if send_len == 0:
110 raise RuntimeError("Connection interruption")

165def get_port(ip):
166 """!
167 @brief 获取数据流端口
168 @details 根据目标IP地址计算出端口号
169 @param ip 目标ip
170 @return 计算的端口号
171 """
172 ip_match = r'^(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|[1-9])\.(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|\d)\.(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|\d)\.(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|\d)$'
173 if len(ip) >= 12 and re.match(ip_match, ip):
174 _port = ip.split('.')[3][-2:]
175 _port = _port[0] + '00' + _port[1]
176 return int(_port)
177 else:
178 return 6001

288 def alloc_buffer(self, length: int, buf: Union[int, np.ndarray, None] = None) -> int:
289 """!
290 @brief 申请一片内存
291 @details 根据传入参数实例化内存类，存入字典中
292 @param length 申请长度
293 @param buf 内存类型
294 @return 申请的内存在字典中的key
295 """
296 # 输入buf为内存指针时
297 if isinstance(buf, int):
298 _memory = np.frombuffer((ctypes.c_uint * length).from_address(buf), dtype='u4')
299 # 输入buf为numpy.ndarray时
300 elif isinstance(buf, np.ndarray):
301 _memory = np.frombuffer(buf, dtype='u4')
302 else:
303 _memory = np.zeros(shape=length, dtype='u4')
304 # 截取所需的内存大小
305 if _memory.size < length:
306 raise ValueError(f'The memory size of the input buf is less than length')
307 _memory = _memory[:length]
308 # 生成Memory对象，在类内描述一片内存
309 memory_obj = self.Memory(memory=_memory, size=length, idx=self.memory_index, using_event=Event())
310 self.memory_dict[self.memory_index] = memory_obj
311 self.memory_index += 1
312 return memory_obj.idx

376 def _recv(self, fd, length, offset, event):
377 """!
378 @brief 数据上行
379 @details 数据流上行的具体实现
380 @param fd 内存标号(key)
381 @param length 要接收数据的长度
382 @param offset 内存偏移量
383 @param event 外部停止信号量
384 @return 已使用内存大小
385 """
386 if fd not in self.memory_dict:
387 raise RuntimeError(f"没有此内存块")
388 memory_object = self.memory_dict[fd]
389 if memory_object.using_event.is_set():
390 raise RuntimeError("内存正在被使用")
391 memory_object.using_event.set()
392 if length > memory_object.size:
393 raise RuntimeError(f"数据大小超过内存大小")
394 if length > (memory_object.size - offset):
395 raise RuntimeError(f"偏移量过大")
396 if length % 4 != 0:
397 raise RuntimeError(f"数据不能被4整除")
398 memory_object.using_size = 0
399 write_len = 0
400 while True:
401 try:
402 if event.is_set():
403 memory_object.using_event.clear()
404 return memory_object.using_size
405 if length*4 > 1024:
406 recv_length = 1024
407 else:
408 recv_length = length*4
409
410 fd = self._recv_server.recv(recv_length)
411 fd_len = len(fd)
412 if fd_len < recv_length:
413 fd += self._recv_server.recv(recv_length-fd_len)
414 fd_len = len(fd)
415 if fd_len == 0:
416 memory_object.using_event.clear()
417 logging.info("Recv complete")
418 break
419
420 # logging.debug(msg=f"Recv data is: {fd.hex()}")
421
422 memory_object.memory[write_len + offset:write_len + offset + fd_len // 4] = np.frombuffer(fd, dtype='u4')
423 write_len += (fd_len // 4)
424
425 memory_object.using_size = memory_object.using_size + (write_len + offset + fd_len // 4) - (
426 write_len + offset)
427
428 if length <= 0:
429 memory_object.using_event.clear()
430 return memory_object.using_size
431
432 length -= (fd_len // 4)
433 except Exception as e:
434 logging.error(msg=e)
435 memory_object.using_event.clear()
436 break
