107 def load(self):
108 """!
109 从文件中加载数据，并相应地设置实例变量。
110 此方法从指定的文件中读取数据，通常文件采用JSON格式存储，并将读取的数据设置为相关实例变量。
111 文件中包含了与接口控制文档 (ICD) 相关的信息。
112
113 @return: 若数据成功加载，则返回True；否则返回False。
114 @rtype: bool
115 """
116 file_path = self._file_name
117 # 使用utf-8编码打开文件并以读取模式
118 with open(file_path, 'r', encoding='utf-8') as fp:
119 try:
120 # 尝试将JSON数据从文件加载到 'icd_data' 实例变量中
121 self.icd_data = json.load(fp)
122 except json.decoder.JSONDecodeError as e:
123 # 如果出现JSON解码错误，记录错误消息并返回False表示加载失败
124 logging.error(msg=f'{e}, {self._file_name} unavailable')
125 return False
126 try:
127 # 从 'icd_data' 字典中提取特定数据并将其设置为实例变量
128 self.param = self.icd_data['param']
129 self.command = self.icd_data['command']
130 self.sequence = self.icd_data['sequence']
131 # 记录成功消息，表示ICD参数加载成功
132 logging.info(msg='ICD Parameters loaded successfully')
133 except Exception as e:
134 # 如果提取数据或设置实例变量时出现异常，记录错误并返回False
135 logging.error(msg=f'{e}, {file_path} unavailable')
136 return False
137 # 返回True，表示数据加载成功
138 return True

155 def get_param(self, param_name: str, default=0, fmt_type=int):
156 """!
157 @brief 获取icd参数
158 @details 根据参数名称获取参数值
159 @param param_name: 参数名称
160 @param default: 默认值
161 @param fmt_type: 参数值格式化类型
162 @return: 格式化后的参数值
163 """
164 param = self.param.get(param_name, None)
165 if param[1] == "file_data" or param[1] == "file_length":
166 return param[2]
167 if param is None:
168 logging.warning(msg=f'未找到参数：{param_name}')
169 self.param.update({param_name: [param_name, 'uint32', default]})
170 return fmt_type(default)
171 return fmt_type(param[2])

173 def set_param(self, param_name: str, value, fmt_type=int):
174 """!
175 @brief 设置参数值
176 @details 根据参数名称设置相应的值
177 @param param_name: 参数名称
178 @param value: 参数值
179 @param fmt_type: 参数值格式化类型
180 @return:
181 """
182 param = self.param.get(param_name, [param_name, 'uint32', value])
183 if isinstance(value, str) and value.startswith('0x'):
184 param[2] = int(value, 16)
185 elif isinstance(value, str) and value.startswith('0b'):
186 param[2] = int(value, 2)
187 elif param[1] == "file_data" or param[1] == "file_length":
188 param[2] = value
189 elif isinstance(value, str) and '.' in value and param[1] != 'file':
190 param[2] = float(value)
191 else:
192 param[2] = value_python[param[1]](value)
193 self.param.update({param_name: param})

195 def fmt_command(self, command_name, command_type: str = "send", file_name=None) -> bytes:
196 """!
197 @brief 格式化指令
198 @details 根据指令名、指令类型和文件名组合成指令
199 @param command_name: 指令名称
200 @param command_type: 指令类型(发送接收)
201 @param file_name: 文件名
202 @return: 格式化好的指令
203 """
204 file_data = b''
205 command = []
206 file_length = 0
207 if isinstance(file_name, str):
208 file_data, file_length = self.__get_file(file_name)
209 try:
210 target_bytes = []
211 if command_name in self.sequence:
212 sequence_data: pd.DataFrame = pd.read_excel(file_name)
213 sequence_cmd = self.sequence[command_name]
214 for row in range(sequence_data.shape[0]):
215 for register in sequence_cmd:
216 if isinstance(register, str):
217 _reg = self.param.get(register, None)
218 assert _reg, f'未找到参数{register}'
219 if register in sequence_data:
220 value = sequence_data[register][row]
221 else:
222 value = _reg[2]
223 else:
224 _reg = register
225 value = _reg[2]
226 value, _fmt = self.__fmt_register(_reg, value)
227 target_bytes.append(struct.pack(self.fmt_mode + _fmt, value))
228
229 for register in self.command[command_name][command_type]:
230 if isinstance(register, list):
231 value, _fmt = self.__fmt_register(register, register[2])
232 if _fmt == 'file_data':
233 command.append(value)
234 else:
235 command.append(struct.pack(self.fmt_mode + _fmt, value))
236 elif isinstance(register, str):
237 if register == file_context_flag:
238 command.append(file_data)
239 elif register == file_length_flag:
240 command.append(struct.pack(self.fmt_mode + 'I', file_length))
241 elif register in self.param:
242 value, _fmt = self.__fmt_register(self.param[register], self.param[register][2])
243 if _fmt == 'file_data':
244 command.append(value)
245 else:
246 command.append(struct.pack(self.fmt_mode + _fmt, value))
247 elif register == f'{{{{{command_name}}}}}':
248 command.extend(target_bytes)
249 else:
250 logging.warning(msg=f'指令({command_name})的({register})不存在')
251 else:
252 logging.warning(msg=f'指令({command_name})的({register})格式不正确')
253 except Exception as e:
254 logging.error(msg=f'{e},指令转码失败')
255
256 command = b''.join(command)
257 assert len(command) >= 16, f'指令({command_name})不正确'
258 # self.kit.interface.send_bytes(command[0: 12] + struct.pack(self.fmt_mode + 'I', len(command)) + command[16:])
259 return command[0: 12] + struct.pack(self.fmt_mode + 'I', len(command)) + command[16:]

261 def __fmt_register(self, register: list, value):
262 try:
263
264 if register[1] == "file_data":
265 file_data, file_length = self.__get_file(value)
266 return file_data, "file_data"
267 if register[1] == "file_length":
268 file_data, file_length = self.__get_file(value)
269 return file_length, 'I'
270 if isinstance(value, str) and value.startswith('0x'):
271 value = int(value, 16)
272 if isinstance(value, str) and value.startswith('0b'):
273 value = int(value, 2)
274 if len(register) > 3:
275 # 发送时做参数计算
276 x = value
277 value = eval(register[-1])
278 fmt_str = value_type[register[1]]
279 return value_python[register[1]](value), fmt_str
280 except Exception as e:
281 logging.error(msg=f'{e},寄存器({register[0]})有误')
282 return 0, 'I'

294 def execute_icd_command(self, parm_name: str) -> list:
295 """!
296 @brief 查找指令并执行
297 @details 根据参数名查找指令并执行指令
298 @param parm_name: 参数名/指令名

302 command_list = []
303 if parm_name in self.command:
304 command_list.append(parm_name)
305 else:
306 for command in self.command:
307 if parm_name in self.command[command]["send"]:
308 command_list.append(command)
309
310 # 发送
311 if self.check_recv_head:
312 return self.send_and_check(command_list)
313 else:
314 return self.send_and_not_check(command_list)

316 def send_and_check(self, command_list):
317 result_list = []
318 for command in command_list:
319 if len(self.command[command]["recv"]) < 5:
320 # 接收包头, id, 序号, 指令长度, 结果参数
321 raise RuntimeError(f"The {command} recv register is not define, or recv register<5.")
322 else:
323 send_cmd = self.fmt_commandfmt_command(command_name=command, command_type="send")
324 recv_cmd = self.fmt_commandfmt_command(command_name=command, command_type="recv")
325 total_len = len(send_cmd)
326 send_len = self.kit.itf_cmd.send_bytes(send_cmd)
327 if total_len != send_len:
328 raise RuntimeError(f"{command} total_len is {total_len}, but just send {send_len}!")
329 recv = self.kit.itf_cmd.recv_bytes(struct.unpack("=I", recv_cmd[12:16])[0])
330 self.check_recv(recv_cmd, recv, command)
331 result_list.append(recv)
332
333 # 写入到参数中
334 length = 16
335 for index, data in enumerate(self.command[command]["recv"]):
336 if index >= 4:
337 if isinstance(data, list):
338 length += type_size[data[1]]
339 elif isinstance(data, str):
340 data_size = type_size[self.param[data][1]]
341 data_type = value_type[self.param[data][1]]
342 self.set_paramset_param(data, struct.unpack(f"={data_type}", recv[length:length+data_size])[0])
343 length += data_size
344 return result_list
