56 def recv_bytes(self, size: int) -> bytes:
57 """!
58 @brief 接收数据
59 @details 使用网络接收指定大小的数据
60 @param size 接收数据的长度
61 @return 接收到的数据
62 """
63 with self.busy_lock:
64 recv_data = b''
65 recv_size = 0
66 while True:
67 if recv_size != size:
68 data = self._tcp_server.recv(size-recv_size)
69 recv_data += data
70 recv_size += len(data)
71 if recv_size >= size:
72 break
73 return recv_data

75 def send_bytes(self, data: bytes) -> int:
76 """!
77 @brief 发送数据
78 @details 使用网络发送数据
79 @param data 要发送的数据
80 @return 发送完成的数据长度
81 """
82 with self.busy_lock:
83 total_len = len(data)
84 total_sendlen = 0
85 while True:
86 send_len = self._tcp_server.send(data)
87 total_sendlen += send_len
88 if total_len == total_sendlen:
89 return total_len
90 if send_len == 0:
91 raise RuntimeError("Connection interruption")

134def get_port(ip):
135 """!
136 @brief 获取数据流端口
137 @details 根据目标IP地址计算出端口号
138 @param ip 目标ip
139 @return 计算的端口号
140 """
141 ip_match = (r'^(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|[1-9])\.(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|\d)\.(1\d{2}|'
142 r'2[0-4]\d|25[0-5]|[1-9]\d|\d)\.(1\d{2}|2[0-4]\d|25[0-5]|[1-9]\d|\d)$')
143 if len(ip) >= 12 and re.match(ip_match, ip):
144 _port = ip.split('.')[3][-2:]
145 _port = _port[0] + '00' + _port[1]
146 return int(_port)
147 else:
148 return 6001

249 def alloc_buffer(self, length: int, buf: Union[int, np.ndarray, None] = None) -> int:
250 """!
251 @brief 申请一片内存
252 @details 根据传入参数实例化内存类，存入字典中
253 @param length 申请长度
254 @param buf 内存类型
255 @return 申请的内存在字典中的key
256 """
257 # 输入buf为内存指针时
258 if isinstance(buf, int):
259 _memory = np.frombuffer((ctypes.c_uint * length).from_address(buf), dtype='u4')
260 # 输入buf为numpy.ndarray时
261 elif isinstance(buf, np.ndarray):
262 _memory = np.frombuffer(buf, dtype='u4')
263 else:
264 _memory = np.zeros(shape=length, dtype='u4')
265 # 截取所需的内存大小
266 if _memory.size < length:
267 raise ValueError(f'The memory size of the input buf is less than length')
268 _memory = _memory[:length]
269 # 生成Memory对象，在类内描述一片内存
270 memory_obj = self.Memory(memory=_memory, size=length, idx=self.memory_index, using_event=Event())
271 self.memory_dict[self.memory_index] = memory_obj
272 self.memory_index += 1
273 return memory_obj.idx

310 def open_recv(self, chnl: int, fd: int, length: int, offset: int = 0) -> None:
311 """!
312 @brief 数据上行开启
313 @details 开启数据流上行
314 @param chnl
315 @param fd 内存标号(key)
316 @param length 要接收数据的长度
317 @param offset 内存偏移量
318 @return True/False
319 """
320 try:
321 if not self.open_flag:
322 raise RuntimeError("You must use open_board first")
323 if self._recv_server is not None and self._recv_thread.is_alive():
324 raise RuntimeError("Too many recv_thread")

326 logging.info(msg=f"{self.__class__.__name__} Client connection")
327 self._recv_server.settimeout(self._timeout)
328 event = self.stop_event
329 self._recv_thread = threading.Thread(target=self._recv, args=(fd, length, offset, event),
330 daemon=True, name='TCP_recv')

332 except Exception as e:

336 def _recv(self, fd, length, offset, event):
337 """!
338 @brief 数据上行
339 @details 数据流上行的具体实现
340 @param fd 内存标号(key)
341 @param length 要接收数据的长度
342 @param offset 内存偏移量
343 @param event 外部停止信号量
344 @return 已使用内存大小
345 """
346 if fd not in self.memory_dict:
347 raise RuntimeError(f"没有此内存块")
348 memory_object = self.memory_dict[fd]
349 if memory_object.using_event.is_set():
350 raise RuntimeError("内存正在被使用")
351 memory_object.using_event.set()
352 if length > memory_object.size:
353 raise RuntimeError(f"数据大小超过内存大小")
354 if length > (memory_object.size - offset):
355 raise RuntimeError(f"偏移量过大")
356 if length % 4 != 0:
357 raise RuntimeError(f"数据不能被4整除")
358 memory_object.using_size = 0
359 write_len = 0
360 while True:
361 try:
362 if event.is_set():
363 memory_object.using_event.clear()
364 return memory_object.using_size
365 if length*4 > 1024:
366 recv_length = 1024
367 else:
368 recv_length = length*4
369
370 fd = self._recv_server.recv(recv_length)
371 fd_len = len(fd)
372 if fd_len < recv_length:
373 fd += self._recv_server.recv(recv_length-fd_len)
374 fd_len = len(fd)
375 if fd_len == 0:
376 memory_object.using_event.clear()
377 logging.info("Recv complete")
378 break
379
380 # logging.debug(msg=f"Recv data is: {fd.hex()}")
381
382 memory_object.memory[write_len + offset:write_len + offset + fd_len // 4] = np.frombuffer(fd, dtype='u4')
383 write_len += (fd_len // 4)
384
385 memory_object.using_size = memory_object.using_size + (write_len + offset + fd_len // 4) - (
386 write_len + offset)
387
388 if length <= 0:
389 memory_object.using_event.clear()
390 return memory_object.using_size
391
392 length -= (fd_len // 4)
393 except Exception as e:
394 logging.error(msg=e)
395 memory_object.using_event.clear()
396 break

398 def wait_stream(self, fd: int, timeout: float = 0.) -> int:
399 """!
400 @brief 等待完成一次dma操作
401 @details 等待所有数据写入内存
402 @param fd 内存标号(key)
403 @param timeout 超时时间
404 @return 已经写入内存中数据的大小
405 """
406 if not self.open_flag:
407 raise RuntimeError("You must use open_board first")
408 if not self._recv_thread.is_alive():
409 raise RuntimeError("recv_thread not alive")
410 if fd not in self.memory_dict:
411 raise RuntimeError(f"没有此内存块")
412 memory = self.memory_dict[fd]
413 try:
414 memory.using_event.wait(timeout)
415 return memory.using_size
416 except RuntimeError as e:
417 logging.error(msg=e)
