32def dispenser(func):
33 """!
34 装饰器，让被装饰的函数根据stream_mode将调用分到指定的实现上
35 @param func: 被装饰的UMiddleware方法
36 @return:
37 """
38 @wraps(func)
39 def wrapper(*args, **kwargs):
40 self: "VirtualStreamMw" = args[0]
41 if self.stream_mode == self.StreamMode.VIRTUAL:
42 _func = func
43 else:
44 _func = getattr(self.kit.itf_stream, func.__name__)
45 return _func(*args, **kwargs)
46
