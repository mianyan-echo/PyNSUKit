26 """!
27 @brief 控制设备的快速开发接口

33 def __init__(self, cmd_itf_class: "UInterfaceMeta" = None, chnl_itf_class: "UInterfaceMeta" = None):
34 """!
35 @code
36 >>> from nsukit import NSUKit, TCPCmdUItf, PCIECmdUItf
37 >>> nsukit = NSUKit(TCPCmdUItf, PCIECmdUItf)
38 >>> ...
39 @endcode
40
41 @brief 初始化接口
42 @details 根据传入的指令接口和数据流接口进行初始化
43 @param cmd_itf_class: 指令类
44 @param chnl_itf_class: 通道类
45 """
46 if cmd_itf_class is None or chnl_itf_class is None:
47 raise RuntimeError(f'Please pass in a subclass of the {UInterface}')
48 self.itf_cmd: "BaseCmdUItf" = cmd_itf_class()
49 self.itf_chnl: "BaseChnlUItf" = chnl_itf_class()
50 self.mw_cmd: "BaseRegMw" = self.CmdMiddleware(self)

53 def start_command(self, target=None, **kwargs) -> None:
54 """!
55 @code
56 >>> from nsukit import NSUKit, TCPCmdUItf, PCIECmdUItf
57 >>> nsukit = NSUKit(TCPCmdUItf, PCIECmdUItf)
58 >>> nsukit.start_command('127.0.0.1', icd_path='~/icd.json')
59 >>> print(nsukit.read('dds0中心频率'))
60 >>> print(nsukit.read(0x00000000))
61 >>> print(nsukit.bulk_read(['dds0中心频率', 0x00000004, 0x00000008]))
62 @endcode

64 @brief 发送指令前准备
65 @details 根据目标设备类型、参数建立链接

68 @return:

82 def write(self, addr, value, execute=True) -> "list | int":
83 """!
84 @brief 写寄存器
85 @details 按照输入的地址、值进行写寄存器
86 @param addr: 寄存器地址
87 @param value: 要写入的值
88 @param execute: 是否执行icd指令
89 @return: 写入结果
90 """
91 if isinstance(addr, str):
92 if not self.mw_cmd.param_is_command(addr):
93 self.mw_cmd.set_param(param_name=addr, value=value)
94 if execute:
95 return self.mw_cmd.execute_icd_command(addr)
96 else:
97 return self.itf_cmd.write(addr, value)
98
99 def read(self, addr) -> int:
100 """!
101 @brief 读寄存器
102 @details 按照输入的地址、值进行读寄存器
103 @param addr: 寄存器地址
104 @return: 读出的值
105 """
106 if isinstance(addr, str):
107 return self.mw_cmd.get_param(addr)
108 else:
109 return self.itf_cmd.read(addr)
110
111 def bulk_write(self, params: dict) -> list:
112 """!
113 @brief 批量写寄存器
114 @details 按照输入的参数字典以地址、值的方式进行批量写寄存器
115 @param params: 参数字典 {key:value,key:value}
116 @return: 每个指令的发送长度 [int, int, int]
117 """
118 send_len = []
119 for param in params:
120 if isinstance(param, str):
121 send_len.append(self.write(param, params.get(param)))
122 else:
123 send_len.append(self.itf_cmd.write(param, params.get(param)))
124 return send_len
125
126 def bulk_read(self, addrs: list) -> list:
127 """!
128 @brief 批量读寄存器
129 @details 按照输入的参数列表批量读取寄存器
130 @param addrs: 地址列表
131 @return: 每个指令的返回值 [value, value, value]
132 """
133 value = []
134 for addr in addrs:
135 value.append(self.read(addr))
136 return value
