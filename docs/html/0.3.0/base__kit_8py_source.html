26 """!
27 用于NSUKit.bulk_xxx方法的枚举类

30
31 **INCREMENT**: 对一个从一个寄存器地址开始，依次遍历指定数量的寄存器
32 """

134 def write(self, addr: Union[int, Iterable[int]], value: Union[bytes, Iterable[bytes]]) -> None:
135 """!
136 @brief 写寄存器
137 @details 按照输入的地址、值进行写寄存器
138 @anchor NSUKit_write
139 @param addr: 寄存器地址
140 @param value: 要写入的值，为一len为4的bytes对象
141 @return: 无返回值，无报错即成功
142
143 ---
144 调用示例
145 @code
146 >>> kit: NSUSoc
147 >>> kit.write(0x10, b'\x00\x00\x00\x00')
148 @endcode
149
150 @code
151 >>> kit: NSUSoc
152 >>> try:
153 >>> kit.write(0x10, b'\x00'*5)
154 >>> except Exception as e:
155 >>> print(f'Failed')
156 @endcode
157 """
158 check_reg_schema(addr, value)
159 if isinstance(addr, int) and isinstance(value, bytes):
160 self.itf_cr.write(addr, value)
161 elif isinstance(addr, abc.Iterable) and isinstance(value, abc.Iterable):
162 self.itf_cr.multi_write(addr, value)

186 def bulk_write(self, base: int, value: bytes, mode: Union[str, BulkMode] = BulkMode.INCREMENT) -> None:
187 """!
188
189 @brief 片写寄存器
190 @details 按照输入的基地址和指定模式，将数据写入寄存器
191 @anchor NSUKit_bulk_write
192 @param base: 基寄存器地址
193 @param value: 要写入的数据，不限长度
194 @param mode: 枚举类型BulkMode中的的值
195 @return: None，无报错即成功
196
197 ---
198 将基地址0x10的寄存器片，重置为0
199 @code
200 >>> kit: NSUSoc
201 >>> kit.bulk_write(0x10, b'\x00'*51)
202 @endcode
203
204 通过地址为0x10的寄存器，传输一段周期数据
205 @code
206 >>> kit: NSUSoc
207 >>> kit.bulk_write(0x10, b'\x00\x01'*51, mode='inc')
208 @endcode
209 """
210 check_reg_schema(addr=base)
211 e_mode = BulkMode(mode)
212 if e_mode is BulkMode.INCREMENT:
213 self.itf_cr.increment_write(base, value)
214 elif e_mode is BulkMode.LOOP:
215 self.itf_cr.loop_write(base, value)

217 def bulk_read(self, base: int, length: int, mode: Union[str, BulkMode] = BulkMode.INCREMENT) -> bytes:
218 """!
219 @brief 片读寄存器
220 @details 按照输入的基地址和指定模式，从寄存器中读出数据
221 @anchor NSUKit_bulk_read
222 @param base: 基寄存器地址
223 @param length: 要片读的数据长度，可为任意长度
224 @param mode: 枚举类型BulkMode中的的值
225 @return: 读取结果，长度为length指定值
226
227 ---
228 从基地址0x10的寄存器开始，依次读回片内寄存器的值
229 @code
230 >>> kit: NSUSoc
231 >>> value = kit.bulk_read(0x10, 51)
232 >>> len(value) == 51
233 @endcode
234 """
235 check_reg_schema(addr=base)
236 e_mode = BulkMode(mode)
237 if e_mode is BulkMode.INCREMENT:
238 return self.itf_cr.increment_read(base, length)
239 elif e_mode is BulkMode.LOOP:
240 return self.itf_cr.loop_read(base, length)

242 def set_param(self, name: str, value: Any) -> None:
243 """!
244 设置指令参数值
245
246 **并不会真的下发到设备中，只是把相应的参数值写入到主机内存中**
247
248 @param name: 所有指令中包含的参数名，具体支持的参数可以查阅板卡配套的说明文档
249 @param value: 要配置的参数值，支持int、float等值类型，文件路径，numpy.ndarray
250 @return: None,执行失败则报错

254 @code
255 >>> kit: NSUSoc

258 >>> kit.execute('波形预置')
259 @endcode
260 """
