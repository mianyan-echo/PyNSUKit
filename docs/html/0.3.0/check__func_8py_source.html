21def head_check(send_cmd: bytes, recv_cmd: bytes):
22 """!
23 @brief 包头检查
24 @details 返回数据包头检查
25 @param send_cmd 发送的指令
26 @param recv_cmd 返回的指令
27 @return 返回指令的总长度
28 """
29 send_head = struct.unpack('=IIII', send_cmd[:16])
30 recv_head = struct.unpack('=IIII', recv_cmd[:16])
31 if recv_head[0] != 0xCFCFCFCF:
32 raise RuntimeError("返回包头错误")
33 if recv_head[1] != send_head[1]:
34 raise RuntimeError("返回ID错误")
35 return recv_head[3]

45def check_reg_schema(addr: Union[int, Iterable[int]], value: Union[bytes, Iterable[bytes], None] = None) -> None:
46 """!
47 检查寄存器地址和值是否合法
48 @param addr: 寄存器地址，或寄存器地址列表
49 @param value: 寄存器地址或寄存器地址
50 @return None
51 """
52 if isinstance(addr, int) and isinstance(value, bytes):
53 if addr < 0:
54 raise InvalidRegisterValueError(f'The value of addr should be greater than 0, not {addr=}.')
55 if len(value) > 4:
56 raise InvalidRegisterValueError(
57 f'The length of the value must be less than or equal to 4, not {len(value)=}.')
58 elif isinstance(addr, int) and value is None:
59 if addr < 0:
60 raise InvalidRegisterValueError(f'The value of addr should be greater than 0, not {addr=}.')
61 elif isinstance(addr, abc.Iterable) and value is None:
62 for _a in addr:
63 check_reg_schema(_a, value)
64 elif isinstance(addr, abc.Iterable) and isinstance(value, abc.Iterable):
65 for _a, _v in zip(addr, value):
66 check_reg_schema(_a, _v)
67 else:
68 raise InvalidRegisterValueError(
69 f'Unsupported combination of parameter types value[{type(value)}, {value}], addr[{type(addr)}, {addr}]')
